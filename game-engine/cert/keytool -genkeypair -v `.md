keytool -genkeypair -v `
  -alias exampleca `
  -dname "CN=exampleCA, OU=Example Org, O=Example Company, L=San Francisco, ST=California, C=US" `
  -keystore exampleca.jks `
  -keypass:env PW `
  -storepass:env PW `
  -keyalg RSA `
  -keysize 4096 `
  -ext KeyUsage:critical="keyCertSign" `
  -ext BasicConstraints:critical="ca:true" `
  -validity 9999
